import pygame import random # إعدادات المراحل (Stages 1-5: التأسيس) pygame.init() SCREEN_WIDTH = 800 SCREEN_HEIGHT = 600 screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("لعبة القطة - الكود الذهبي 20 مرحلة") # الألوان (Stage 4: Feature Engineering) WHITE = (255, 255, 255) BLACK = (0, 0, 0) CAT_COLOR = (255, 165, 0) # برتقالي # كلاس القطة (Stages 6-10: المنطق والتحليل) class Cat: def __init__(self): self.rect = pygame.Rect(400, 500, 50, 50) self.speed = 7 def move(self, keys): if keys[pygame.K_LEFT] and self.rect.left > 0: self.rect.x -= self.speed if keys[pygame.K_RIGHT] and self.rect.right < SCREEN_WIDTH: self.rect.x += self.speed # نظام النقاط والمراحل (Stages 11-15: الربط والواجهة) class GameSystem: def __init__(self): self.stage = 1 self.score = 0 self.font = pygame.font.SysFont("Arial", 24) def update_stage(self): # كل 10 نقاط ننتقل لمرحلة جديدة (حتى نصل لـ 20) self.stage = (self.score // 10) + 1 if self.stage > 20: self.stage = 20 def draw(self, surface): text = self.font.render(f"Stage: {self.stage}/20 | Score: {self.score}", True, BLACK) surface.blit(text, (10, 10)) # الدورة الرئيسية للعبة (Stages 16-20: الاختبار والتدشين) def start_game(): cat = Cat() system = GameSystem() food = pygame.Rect(random.randint(0, 750), 0, 30, 30) clock = pygame.time.Clock() running = True while running: screen.fill(WHITE) keys = pygame.key.get_pressed() for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # تحريك القطة cat.move(keys) # تحريك الطعام (يزداد الصعوبة مع تقدم المراحل) food.y += 5 + system.stage # تزداد السرعة حسب المرحلة if food.y > SCREEN_HEIGHT: food.y = 0 food.x = random.randint(0, 750) # التحقق من الاصطدام (نقطة) if cat.rect.colliderect(food): system.score += 2 system.update_stage() food.y = 0 food.x = random.randint(0, 750) # رسم العناصر pygame.draw.rect(screen, CAT_COLOR, cat.rect) # القطة pygame.draw.ellipse(screen, (200, 0, 0), food) # الطعام system.draw(screen) pygame.display.flip() clock.tick(60) pygame.quit() # تشغيل المرحلة 20 (التدشين النهائي) if __name__ == "__main__": start_game()
